    2.3.3 еӨҙж–Үд»¶жҗңзҙўи·Ҝеҫ„е…ЁжҷҜ¶

    жүҖеұһз« иҠӮпјҡз¬¬2з«  дәӨеҸүзј–иҜ‘е·Ҙе…·й“ҫ > 2.3 е·Ҙе…·й“ҫзҡ„ж ёеҝғжҰӮеҝө

    йҡҫеәҰпјҡ[IвҶ’I] | йў„и®Ўйҳ…иҜ»ж—¶й—ҙпјҡ12еҲҶй’ҹ

    жң¬иҠӮеҜјиҜ»¶

    жң¬иҠӮжҸӯејҖ GCC жҹҘжүҫеӨҙж–Үд»¶зҡ„е®Ңж•ҙи·Ҝеҫ„жё…еҚ•пјҡд»Һ-IжҢҮе®ҡзҡ„иҮӘе®ҡд№үзӣ®еҪ•еҲ° sysroot дёӢзҡ„зі»з»ҹи·Ҝеҫ„пјҢе№¶ж•ҷдҪ з”ЁдёҖжқЎе‘Ҫд»Өжү“еҚ°еҮәзј–иҜ‘еҷЁ"и„‘еӯҗйҮҢ"зҡ„е®һйҷ…жҗңзҙўеҲ—иЎЁгҖӮ
    еӯҰе®ҢдҪ иғҪзӢ¬з«Ӣи§ЈеҶі 90% зҡ„"fatal error: xxx.h: No such file or directory"й—®йўҳгҖӮ

    GCC еӨҙж–Үд»¶жҗңзҙўи·Ҝеҫ„ [I]¶

    еҪ“дҪ еҶҷдёӢдёҖиЎҢ #include <stdio.h>пјҢGCC е№¶дёҚдјҡж»ЎзЎ¬зӣҳд№ұжүҫгҖӮе®ғжңүдёҖеј еӣәе®ҡзҡ„жҗңзҙўйЎәеәҸиЎЁпјҢжҢүдјҳе…Ҳзә§дҫқж¬ЎжҹҘжүҫпјҢжүҫеҲ°еҚіжӯўгҖӮ

    жҗңзҙўдјҳе…Ҳзә§жҖ»и§Ҳ¶

    GCC зҡ„еӨҙж–Үд»¶жҗңзҙўйҒөеҫӘдёҘж јзҡ„дјҳе…Ҳзә§йЎәеәҸпјҢеҸҜд»Ҙз”ЁдёӢйқўзҡ„жөҒзЁӢзҗҶи§Јпјҡ

    еӣҫ1пјҡGCC еӨҙж–Үд»¶жҗңзҙўжөҒзЁӢеӣҫ: еҸҢеј•еҸ·е’Ңе°–жӢ¬еҸ·зҡ„жҗңзҙўзӯ–з•ҘдёҚеҗҢпјҢ-Iдјҳе…Ҳзә§жңҖй«ҳпјҢsysroot и·Ҝеҫ„жҳҜдәӨеҸүзј–иҜ‘зҡ„е…ій”®гҖӮ

    еҗ„дјҳе…Ҳзә§иҜҰи§Ј¶

    дјҳе…Ҳзә§ и·Ҝеҫ„жқҘжәҗ иҜҙжҳҺ дәӨеҸүзј–иҜ‘дёӯзҡ„зү№ж®ҠжҖ§
    1 -I/dir жҢҮе®ҡзҡ„зӣ®еҪ• з”ЁжҲ·иҮӘе®ҡд№үи·Ҝеҫ„пјҢдјҳе…Ҳзә§жңҖй«ҳ еёёз”ЁдәҺз¬¬дёүж–№еә“еӨҙж–Үд»¶
    2 C_INCLUDE_PATH зҺҜеўғеҸҳйҮҸ йҖҡиҝҮзҺҜеўғеҸҳйҮҸиҝҪеҠ  дёҚжҺЁиҚҗпјҢжҳ“жұЎжҹ“е…ЁеұҖзҺҜеўғ
    3 --sysroot/usr/include дәӨеҸүзј–иҜ‘еҷЁзҡ„"иҷҡжӢҹж №зӣ®еҪ•" дәӨеҸүзј–иҜ‘ж—¶еҸ–д»Ј /usr/include
    4 зј–иҜ‘еҷЁеҶ…зҪ®зі»з»ҹи·Ҝеҫ„ зј–иҜ‘еҷЁзј–иҜ‘ж—¶зЎ¬зј–з Ғзҡ„и·Ҝеҫ„ еҜ№дәӨеҸүзј–иҜ‘еҷЁжҢҮеҗ‘ sysroot еҶ…йғЁ

    вҡ пёҸ йҷ·йҳұпјҡ-I зҡ„йЎәеәҸеҫҲйҮҚиҰҒпјҒGCC жҢүд»Һе·ҰеҲ°еҸізҡ„йЎәеәҸжҗңзҙў -I зӣ®еҪ•пјҢеҰӮжһңеӨҡдёӘзӣ®еҪ•дёӯжңүеҗҢеҗҚеӨҙж–Үд»¶пјҢе…ҲжүҫеҲ°зҡ„йӮЈдёӘдјҡиў«дҪҝз”ЁпјҢеҗҺйқўзҡ„еҚідҪҝзүҲжң¬жӣҙж–°д№ҹдјҡиў«йқҷй»ҳеҝҪз•ҘгҖӮ

    е®һж“Қпјҡжү“еҚ°зј–иҜ‘еҷЁзҡ„зңҹе®һжҗңзҙўи·Ҝеҫ„¶

    GCC еҶ…йғЁи—Ҹеҫ—еҫҲж·ұзҡ„жҗңзҙўи·Ҝеҫ„пјҢеҸҜд»Ҙз”ЁдёҖжқЎе‘Ҫд»Ө"иҜұйӘ—"е®ғеҗҗеҮәжқҘпјҡ

    # еҜ№дәҺдәӨеҸүзј–иҜ‘еҷЁ
$ aarch64-linux-gnu-gcc -xc -E -v /dev/null

# еҜ№дәҺжң¬ең° GCC
$ gcc -xc -E -v /dev/null

    иҝҷжқЎе‘Ҫд»Өзҡ„еҗ«д№үжӢҶи§Јпјҡ

    • -xcпјҡжҠҠиҫ“е…ҘеҪ“дҪң C иҜӯиЁҖеӨ„зҗҶпјҲеҚідҪҝ /dev/null жІЎжңүеҗҺзјҖпјү
    • -EпјҡеҸӘиҝҗиЎҢйў„еӨ„зҗҶйҳ¶ж®өпјҲжӯЈжҳҜжүҫеӨҙж–Үд»¶зҡ„йҳ¶ж®өпјү
    • -vпјҡжҳҫзӨәиҜҰз»ҶиҝҮзЁӢпјҲverboseпјү
    • /dev/nullпјҡз©әиҫ“е…Ҙж–Үд»¶пјҢжҲ‘д»¬еҸӘжҳҜиҰҒи§ҰеҸ‘йў„еӨ„зҗҶжқҘ"зңӢиҝҮзЁӢ"

    еңЁиҫ“еҮәдёӯпјҢжүҫеҲ°иҝҷдёӨж®өе…ій”®еҶ…е®№пјҡ

    #include "..." search starts here:

    рҹ’Ў жҸҗзӨәпјҡдәӨеҸүзј–иҜ‘еҷЁзҡ„иҫ“еҮәдёҚдјҡеҮәзҺ° /usr/includeпјҲе®ҝдё»жңә x86 еӨҙж–Үд»¶зӣ®еҪ•пјүпјҢеҸ–иҖҢд»Јд№Ӣзҡ„жҳҜ /usr/arm-linux-gnueabihf/include жҲ– sysroot дёӢзҡ„и·Ҝеҫ„гҖӮ
    еҰӮжһңеҮәзҺ° /usr/includeпјҢиҜҙжҳҺдҪ зҡ„дәӨеҸүзј–иҜ‘еҷЁй…ҚзҪ®жңүй—®йўҳпјҒ

    йӘҢиҜҒ sysroot и·Ҝеҫ„жҳҜеҗҰз”ҹж•Ҳ¶

    з»“еҗҲдёҠдёҖиҠӮпјҲ2.3.1пјүзҡ„зҹҘиҜҶпјҢеҸҜд»ҘйӘҢиҜҒ sysroot жҳҜеҗҰиў«жӯЈзЎ®зәіе…Ҙжҗңзҙўпјҡ

    SYSROOT=$(arm-linux-gnueabihf-gcc -print-sysroot)
arm-linux-gnueabihf-gcc -xc -E -v /dev/null 2>&1 | grep -A5 "search starts here"

# зЎ®и®Ө sysroot/usr/include еңЁеҲ—иЎЁдёӯ
ls -d "$SYSROOT/usr/include"

    еёёи§Ғ"еӨҙж–Үд»¶жүҫдёҚеҲ°"жҺ’й”ҷ [I]¶

    еөҢе…ҘејҸејҖеҸ‘дёӯпјҢ"fatal error: xxx.h: No such file or directory"жҳҜжңҖи®©дәәжҠ“зӢӮзҡ„й”ҷиҜҜд№ӢдёҖгҖӮдёӢйқўдёүдёӘеңәжҷҜиҰҶзӣ–дәҶ 90% зҡ„зңҹе®һжЎҲдҫӢгҖӮ

    еңәжҷҜ1пјҡ-I еҝҳдәҶеҠ жҲ–и·Ҝеҫ„еҶҷй”ҷ¶

    зҺ°иұЎпјҡ 

    fatal error: mylib.h: No such file or directory
   3 | #include "mylib.h"

    иҜҠж–ӯпјҡеӨҙж–Үд»¶еңЁеҪ“еүҚзӣ®еҪ•жҲ–жҹҗдёӘеӯҗзӣ®еҪ•йҮҢпјҢдҪҶдҪ жІЎе‘ҠиҜү GCC еҺ»е“ӘйҮҢжүҫгҖӮ

    дҝ®еӨҚпјҡ 

    # еҰӮжһң mylib.h еңЁ ./include/ дёӢ
gcc main.c -I./include -o main

# еҰӮжһң mylib.h еңЁ ../common/inc/ дёӢ
gcc main.c -I../common/inc -o main

    вҡ пёҸ йҷ·йҳұпјҡ-I зҡ„йЎәеәҸеҫҲйҮҚиҰҒпјҒGCC жҢүд»Һе·ҰеҲ°еҸізҡ„йЎәеәҸжҗңзҙў -I зӣ®еҪ•пјҢеҰӮжһңеӨҡдёӘзӣ®еҪ•дёӯжңүеҗҢеҗҚеӨҙж–Үд»¶пјҢе…ҲжүҫеҲ°зҡ„йӮЈдёӘдјҡиў«дҪҝз”ЁпјҢеҗҺйқўзҡ„еҚідҪҝзүҲжң¬жӣҙж–°д№ҹдјҡиў«йқҷй»ҳеҝҪз•ҘгҖӮMakefile дёӯеҰӮжһңи·Ҝеҫ„д»ҘеҸҳйҮҸдј е…ҘпјҲеҰӮ -I$(INC_DIR)пјүпјҢиҜ·еҠЎеҝ…зЎ®и®ӨеҸҳйҮҸе·Іе®ҡд№үдё”дёҚдёәз©әгҖӮ

    еңәжҷҜ2пјҡеӨҙж–Үд»¶дёҚеңЁ sysroot йҮҢ¶

    зҺ°иұЎпјҡдәӨеҸүзј–иҜ‘ж—¶жҸҗзӨәжүҫдёҚеҲ°ж ҮеҮҶеӨҙж–Үд»¶пјҢеҰӮ stdio.hгҖӮ

    иҜҠж–ӯпјҡдҪ зҡ„дәӨеҸүзј–иҜ‘еҷЁ sysroot жҳҜжңҖе°ҸеҢ–зҡ„пјҢеҸҜиғҪеҸӘеҢ…еҗ«еҹәзЎҖ libc еӨҙж–Үд»¶гҖӮз¬¬дёүж–№еә“пјҲеҰӮ libxml2гҖҒopensslпјүзҡ„еӨҙж–Үд»¶йңҖиҰҒйўқеӨ–е®үиЈ…еҲ° sysroot дёӯгҖӮ

    дҝ®еӨҚпјҡ 

    # жҹҘзңӢ sysroot йҮҢеҲ°еә•жңүе“ӘдәӣеӨҙж–Үд»¶
SYSROOT=$(arm-linux-gnueabihf-gcc -print-sysroot)
ls "$SYSROOT/usr/include/" | grep -i xml
# еҰӮжһңдёәз©әпјҢиҜҙжҳҺ libxml2 еӨҙж–Үд»¶жІЎиЈ…

    рҹ’Ў жҸҗзӨәпјҡеңЁеөҢе…ҘејҸејҖеҸ‘дёӯпјҢжӯЈзЎ®зҡ„еҒҡжі•жҳҜжҠҠз¬¬дёүж–№еә“дәӨеҸүзј–иҜ‘е®үиЈ…еҲ° sysrootпјҢиҖҢдёҚжҳҜз”Ё -I жҢҮеҗ‘жәҗз Ғзӣ®еҪ•гҖӮиҝҷж ·й“ҫжҺҘйҳ¶ж®өд№ҹиғҪжүҫеҲ°еҜ№еә”зҡ„еә“ж–Үд»¶гҖӮ

    еңәжҷҜ3пјҡдәӨеҸүзј–иҜ‘еҷЁеҒ·еҒ·жүҫдәҶ PC зҡ„еӨҙж–Үд»¶ [рҹ”ҙ з»Ҹе…ёйҷ·йҳұ]¶

    зҺ°иұЎпјҡзЁӢеәҸзј–иҜ‘йҖҡиҝҮдәҶпјҢдҪҶеңЁзӣ®ж ҮжқҝдёҠиҝҗиЎҢејӮеёёпјӣжҲ–иҖ…зј–иҜ‘ж—¶жҸҗзӨәжҹҗдәӣз»“жһ„дҪ“/е®Ҹе®ҡд№үдёҚеӯҳеңЁпјҢдҪҶдҪ зҹҘйҒ“зӣ®ж Үзі»з»ҹеә”иҜҘжңүгҖӮ

    иҜҠж–ӯпјҡзј–иҜ‘еҷЁж··е…ҘдәҶе®ҝдё»жңәзҡ„еӨҙж–Үд»¶пјҒиҝҷйҖҡеёёеҸ‘з”ҹеңЁд»ҘдёӢжғ…еҶөпјҡ

    1. дҪ жүӢеҠЁеҠ дәҶ -I/usr/include
    2. зј–иҜ‘еҷЁжІЎжңүжӯЈзЎ®й…ҚзҪ® sysrootпјҢеӣһйҖҖеҲ°дәҶе®ҝдё»жңәи·Ҝеҫ„
    3. жҹҗдёӘ Makefile зЎ¬зј–з ҒдәҶз»қеҜ№и·Ҝеҫ„

    йӘҢиҜҒпјҡ 

    # жү“еҚ°е®һйҷ…жҗңзҙўи·Ҝеҫ„пјҢжЈҖжҹҘжҳҜеҗҰжңү /usr/include
arm-linux-gnueabihf-gcc -xc -E -v /dev/null 2>&1 | grep "/usr/include"

# еҰӮжһңжңүиҫ“еҮәпјҢиҜҙжҳҺзЎ®е®һеңЁжҗңзҙў PC зҡ„еӨҙж–Үд»¶пјҒ

    дҝ®еӨҚпјҡ

    • еҲ йҷӨ Makefile дёӯзЎ¬зј–з Ғзҡ„ -I/usr/include
    • з”Ё --sysroot= ејәеҲ¶жҢҮе®ҡжӯЈзЎ®зҡ„ sysroot
    • з”Ё -nostdinc зҰҒз”ЁжүҖжңүж ҮеҮҶи·Ҝеҫ„пјҢеҸӘдҝқз•ҷ -I жҢҮе®ҡзҡ„и·Ҝеҫ„пјҲй«ҳзә§з”Ёжі•пјү

    рҹ”ҙ еҚұйҷ©пјҡз”Ё x86 зҡ„еӨҙж–Үд»¶зј–иҜ‘ ARM зЁӢеәҸжҳҜйқҷй»ҳжқҖжүӢгҖӮ е®ғдёҚдјҡжҠҘй”ҷпјҢдҪҶдјҡеҜјиҮҙз»“жһ„дҪ“еӨ§е°ҸгҖҒеӯ—иҠӮеәҸгҖҒе®Ҹе®ҡд№үе…ЁйғЁй”ҷд№ұпјҢжңҖз»Ҳдә§зү©еңЁжқҝеӯҗдёҠеҙ©жәғжҲ–иЎҢдёәејӮеёёгҖӮеҠЎеҝ…з”Ё -v е®ҡжңҹжЈҖжҹҘжҗңзҙўи·Ҝеҫ„пјҒ

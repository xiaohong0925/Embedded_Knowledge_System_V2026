### <strong>数据传输的核心开销来自“系统调用上下文切换”和“内存拷贝”——传统read/write接口单次传输小数据时，上下文切换耗时占比超60%；多轮拷贝（如“设备→内核→用户→网络”）会导致带宽利用率不足。
优化目标是“减少系统调用次数”和“消除冗余内存拷贝”。</strong>


### <strong>批量操作：减少系统调用次数的设计技巧</strong>

系统调用的本质是“用户态与<span class="red">内核态</span>的权限切换”，单次切换耗时约1-5μs，若高频传输小数据（如100字节/次），每秒10万次调用会占用50%以上CPU资源。批量操作通过“一次系统调用传输多组数据”降低调用频率，核心适用于“高频小数据传输场景”（如工业传感器周期性采样、物联网设备状态上报）。

###### 1.1.1 性能瓶颈分析（以SHT30传感器为例）
传统单次读取方案（每秒1000次采样）的性能瓶颈：
| 指标                | 单次读取（传统方案） | 问题分析                          |
|---------------------|----------------------|-----------------------------------|
| 系统调用次数        | 1000次/秒            | 上下文切换频繁，CPU占用率高       |
| 单次调用数据量      | 8字节（温湿度+时间戳） | 数据量小，调用开销占比超80%       |
| CPU占用率（单核）   | 35%                  | 频繁切换导致CPU资源浪费           |
| 最大支持采样率      | 2000次/秒            | 超过后CPU占满，出现采样丢点       |

###### 1.1.2 批量操作实战：驱动与应用层改造
以SHT30传感器为例，实现“一次read调用读取10组数据”的批量传输方案，核心是驱动层引入“批量缓冲区”，应用层按批量格式解析。

1.  **驱动层改造（新增批量缓存与读取逻辑）**：
   ```c
   #include <linux/module.h>
   #include <linux/fs.h>
   #include <linux/cdev.h>
   #include <linux/i2c.h>
   #include <linux/spinlock.h>

   // 传感器单组数据结构（8字节）
   struct sensor_data {
       float temp;    // 温度（4字节）
       float humi;    // 湿度（4字节）
   };

   // 批量数据结构（10组数据+头部标识，共81字节）
   #define BATCH_COUNT 10  // 批量传输组数
   struct batch_data {
       uint8_t magic;               // 魔术字（0xAA，用于校验）
       struct sensor_data data[BATCH_COUNT];  // 批量数据
       uint8_t crc;                 // 整体CRC校验（可选，保障批量数据完整性）
   };

   // 自定义设备结构体（新增批量缓冲区）
   struct sensor_dev {
       struct i2c_client *client;
       struct cdev cdev;
       dev_t dev_num;
       struct batch_data batch_buf;  // 批量缓冲区
       int batch_idx;                // 批量缓冲区当前填充索引
       spinlock_t batch_lock;        // 自旋锁（保护批量缓冲区并发访问）
   };
   static struct sensor_dev sht30_dev;

   // 周期性采集函数（填充批量缓冲区）
   static void sensor_collect(struct work_struct *work) {
       struct sensor_dev *dev = container_of(work, struct sensor_dev, work.work);
       struct sensor_data single_data;
       unsigned long flags;

       // 1. 采集单组数据（I2C读取+CRC校验，逻辑同前）
       single_data.temp = 25.5 + (rand()%10)/10.0;  // 模拟采集数据
       single_data.humi = 45.0 + (rand()%10)/10.0;

       // 2. 填充批量缓冲区（自旋锁保护）
       spin_lock_irqsave(&dev->batch_lock, flags);
       if (dev->batch_idx < BATCH_COUNT) {
           dev->batch_buf.data[dev->batch_idx++] = single_data;
           // 缓冲区填满时设置魔术字和CRC
           if (dev->batch_idx == BATCH_COUNT) {
               dev->batch_buf.magic = 0xAA;
               dev->batch_buf.crc = crc8(&dev->batch_buf, sizeof(struct batch_data)-1);
           }
       }
       spin_unlock_irqrestore(&dev->batch_lock, flags);

       // 3. 10ms后再次采集（对应100Hz采样率，批量10组对应100ms传输一次）
       schedule_delayed_work(&dev->work, msecs_to_jiffies(10));
   }

   // 批量read接口实现
   static ssize_t sensor_read(struct file *filp, char __user *buf, size_t count, loff_t *ppos) {
       struct sensor_dev *dev = filp->private_data;
       ssize_t ret;
       unsigned long flags;

       // 1. 校验读取长度（必须是批量数据结构大小）
       if (count != sizeof(struct batch_data)) {
           dev_err(&dev->client->dev, "Read len err! req=%zu, need=%zu\n",
                   count, sizeof(struct batch_data));
           return -EINVAL;
       }

       // 2. 等待批量缓冲区填满（阻塞模式，用等待队列实现，简化未写）
       if (dev->batch_idx < BATCH_COUNT) {
           wait_event_interruptible(dev->wait_q, dev->batch_idx >= BATCH_COUNT);
       }

       // 3. 拷贝批量数据到用户态（自旋锁保护，避免采集过程中被修改）
       spin_lock_irqsave(&dev->batch_lock, flags);
       ret = copy_to_user(buf, &dev->batch_buf, sizeof(struct batch_data));
       if (ret == 0) {
           dev->batch_idx = 0;  // 拷贝完成后重置缓冲区索引
           ret = sizeof(struct batch_data);
       } else {
           ret = -EFAULT;
       }
       spin_unlock_irqrestore(&dev->batch_lock, flags);

       return ret;
   }
   ```

2.  **应用层改造（批量读取与解析）**：
   ```c
   #include <stdio.h>
   #include <fcntl.h>
   #include <unistd.h>
   #include <string.h>

   #define BATCH_COUNT 10
   struct sensor_data {
       float temp;
       float humi;
   };
   struct batch_data {
       uint8_t magic;
       struct sensor_data data[BATCH_COUNT];
       uint8_t crc;
   };

   // CRC校验函数（与驱动一致）
   static uint8_t crc8(uint8_t *data, size_t len) {
       uint8_t crc = 0xFF;
       for (int i=0; i<len; i++) {
           crc ^= data[i];
           for (int j=0; j<8; j++) {
               crc = (crc << 1) ^ ((crc & 0x80) ? 0x31 : 0);
           }
       }
       return crc;
   }

   int main(void) {
       int fd, ret;
       struct batch_data batch;
       fd = open("/dev/sht30_sensor", O_RDONLY);
       if (fd < 0) { perror("open"); return -1; }

       // 循环批量读取（一次读10组数据，对应100ms）
       while (1) {
           ret = read(fd, &batch, sizeof(batch));
           if (ret != sizeof(batch)) {
               perror("read");
               continue;
           }

           // 校验批量数据完整性（魔术字+CRC）
           if (batch.magic != 0xAA || crc8((uint8_t*)&batch, sizeof(batch)-1) != batch.crc) {
               fprintf(stderr, "Batch data corrupted!\n");
               continue;
           }

           // 解析并打印批量数据
           printf("Batch read success (%d samples):\n", BATCH_COUNT);
           for (int i=0; i<BATCH_COUNT; i++) {
               printf("Sample %d: temp=%.1f℃, humi=%.1f%%RH\n",
                      i+1, batch.data[i].temp, batch.data[i].humi);
           }
       }
       close(fd);
       return 0;
   }
   ```

### <strong>零拷贝技术：sendfile与splice在接口中的应用 [M]</strong>

传统数据传输（如“摄像头→内核→用户→网络”）存在**4次内存拷贝**（设备→内核缓冲区、内核→用户缓冲区、用户→socket缓冲区、socket→网卡）和**3次上下文切换**，零拷贝技术通过“<span class="red">内核态</span>直接数据搬运”消除用户态与内核态的拷贝，大师级优化需掌握“sendfile”（文件到网络）和“splice”（任意两个文件描述符）的适用场景与实战。

###### 1.2.1 零拷贝核心原理与技术对比
| 传输场景               | 传统方案（read+write）       | sendfile（零拷贝）              | splice（零拷贝）                |
|------------------------|-------------------------------|---------------------------------|---------------------------------|
| 数据路径               | 内核→用户→内核               | 内核直接搬运（无用户态参与）    | 内核直接搬运（管道作为中间层）  |
| 内存拷贝次数           | 4次                           | 2次（设备→内核→网卡，部分硬件支持1次） | 2次                             |
| 上下文切换次数         | 4次                           | 2次                             | 2次                             |
| 适用场景               | 任意数据传输                 | 文件→网络（如视频推流、日志上传） | 任意两个fd（如内核缓冲区→网络） |
| 用户态是否可修改数据   | 是                           | 否（内核态直接传输）            | 否（需通过管道中转）            |

###### 1.2.2 实战1：sendfile实现摄像头图像网络推流
场景：将OV5640摄像头采集的图像（存储为`/dev/ov5640_frame`字符设备）通过TCP推流到服务器，传统方案用read+write，优化方案用sendfile消除<span class="red">用户态</span>拷贝。

1.  **驱动层准备（字符设备模拟图像帧）**：
   摄像头驱动需将采集的图像帧写入内核缓冲区，并暴露为字符设备`/dev/ov5640_frame`，核心是确保缓冲区为“可被sendfile访问的页缓存”或“<span class="red">DMA</span>缓冲区”。

2.  **应用层sendfile推流实现**：
   ```c
   #include <stdio.h>
   #include <fcntl.h>
   #include <sys/socket.h>
   #include <netinet/in.h>
   #include <sys/sendfile.h>
   #include <sys/stat.h>
   #include <unistd.h>

   #define SERVER_IP "192.168.1.101"
   #define SERVER_PORT 8080
   #define FRAME_SIZE (1920*1080*2)  // 1080P YUV422格式，4.1MB/帧

   int main(void) {
       int fd, sockfd, ret;
       struct sockaddr_in server_addr;
       struct stat st;

       // 1. 打开摄像头帧设备（内核缓冲区）
       fd = open("/dev/ov5640_frame", O_RDONLY);
       if (fd < 0) { perror("open frame"); return -1; }
       fstat(fd, &st);  // 获取设备文件信息（sendfile需文件长度）

       // 2. 创建TCP连接到服务器
       sockfd = socket(AF_INET, SOCK_STREAM, 0);
       server_addr.sin_family = AF_INET;
       server_addr.sin_port = htons(SERVER_PORT);
       inet_pton(AF_INET, SERVER_IP, &server_addr.sin_addr);
       ret = connect(sockfd, (struct sockaddr*)&server_addr, sizeof(server_addr));
       if (ret < 0) { perror("connect"); return -1; }

       // 3. 零拷贝推流（sendfile：内核直接从帧设备→socket）
       off_t offset = 0;
       printf("Start push stream (zero-copy)...\n");
       while (1) {
           // sendfile参数：sockfd（目标）、fd（源）、偏移量、长度
           ret = sendfile(sockfd, fd, &offset, FRAME_SIZE);
           if (ret < 0) {
               perror("sendfile");
               break;
           }
           printf("Push frame: %d bytes, offset=%lld\n", ret, (long long)offset);
           offset = 0;  // 重置偏移量（循环读取帧）
           usleep(33000);  // 30帧/秒（1000ms/30≈33ms）
       }

       close(fd);
       close(sockfd);
       return 0;
   }
   ```

###### 1.2.3 实战2：splice实现内核缓冲区到本地文件的零拷贝存储
场景：将SHT30传感器的批量数据（内核缓冲区）直接存储到本地日志文件，splice支持“任意两个文件描述符”的零拷贝传输，需用管道作为中间层。

1.  **应用层splice实现**：
   ```c
   #include <stdio.h>
   #include <fcntl.h>
   #include <unistd.h>
   #include <sys/splice.h>

   #define PIPE_SIZE 65536  // 管道缓冲区大小
   #define BATCH_SIZE sizeof(struct batch_data)  // 批量数据大小

   int main(void) {
       int fd_sensor, fd_log, pipe_fd[2];
       ssize_t ret;

       // 1. 创建管道（splice需管道作为中间层）
       if (pipe(pipe_fd) < 0) { perror("pipe"); return -1; }

       // 2. 打开传感器设备和日志文件
       fd_sensor = open("/dev/sht30_sensor", O_RDONLY);
       fd_log = open("sensor_log.bin", O_WRONLY | O_CREAT | O_TRUNC, 0644);
       if (fd_sensor < 0 || fd_log < 0) { perror("open"); return -1; }

       // 3. splice零拷贝传输：传感器→管道→日志文件
       printf("Start zero-copy save (splice)...\n");
       while (1) {
           // 第一步：传感器→管道（零拷贝）
           ret = splice(fd_sensor, NULL, pipe_fd[1], NULL,
                       BATCH_SIZE, SPLICE_F_MOVE | SPLICE_F_NONBLOCK);
           if (ret < 0) { perror("splice sensor->pipe"); continue; }

           // 第二步：管道→日志文件（零拷贝）
           ret = splice(pipe_fd[0], NULL, fd_log, NULL,
                       ret, SPLICE_F_MOVE | SPLICE_F_NONBLOCK);
           if (ret < 0) { perror("splice pipe->log"); continue; }

           printf("Saved batch data: %zd bytes\n", ret);
       }

       close(fd_sensor);
       close(fd_log);
       close(pipe_fd[0]);
       close(pipe_fd[1]);
       return 0;
   }
   ```

2.  **大师级注意事项**：
   - sendfile限制：仅支持“文件描述符指向真实文件”（如字符设备、普通文件），不支持socket作为源（如从网络接收后转发）；
   - splice限制：必须通过管道作为中间层，且两个文件描述符中至少有一个是管道；
   - 硬件零拷贝：部分高端SoC（如NXP i.MX8）支持“<span class="red">DMA</span>直接从设备到网卡”（1次拷贝），需配合驱动层DMA通道配置。<br>

### <strong>嵌入式系统资源（内存、文件描述符）有限，接口若存在“内存泄漏”“句柄泄漏”，会导致系统运行一段时间后崩溃——工业设备通常要求7×24小时运行，资源泄漏是致命缺陷。优化目标是“杜绝泄漏”和“高效复用资源”。</strong>


### <strong>内存泄漏防护：kmalloc与kfree的配对检查方法</strong>

内存泄漏是驱动开发中最隐蔽的问题，多源于“分支逻辑中分配内存后未释放”“模块卸载时未回收资源”。防护需结合“编码规范”“静态检查”“动态检测”三重手段，大师级技巧是熟练使用内核自带的kmemleak工具定位泄漏点。

###### 2.1.1 泄漏场景复现（驱动层常见错误）
```c
// 泄漏代码：分支中分配内存后未释放
static int sensor_probe(struct i2c_client *client) {
    struct sensor_dev *dev = kzalloc(sizeof(struct sensor_dev), GFP_KERNEL);
    if (!dev) return -ENOMEM;

    // 分支1：初始化I2C失败，未释放dev
    if (i2c_set_clientdata(client, dev) < 0) {
        dev_err(&client->dev, "Set client data failed!\n");
        return -EIO;  // 泄漏点：dev未释放
    }

    // 分支2：初始化等待队列失败，未释放dev
    init_waitqueue_head(&dev->wait_q);
    if (alloc_chrdev_region(&dev->dev_num, 0, 1, "sht30") < 0) {
        dev_err(&client->dev, "Alloc dev num failed!\n");
        return -ENODEV;  // 泄漏点：dev未释放
    }

    return 0;
}

// 泄漏代码：模块卸载时未释放批量缓冲区
static void sensor_remove(struct i2c_client *client) {
    struct sensor_dev *dev = i2c_get_clientdata(client);
    cdev_del(&dev->cdev);
    unregister_chrdev_region(&dev->dev_num, 1);
    // 泄漏点：未释放dev->batch_buf（若为动态分配）
    kfree(dev);  // 仅释放dev，未释放成员变量
}
```

###### 2.1.2 防护方案：编码规范与静态检查
1.  **编码规范：遵循“分配即释放”原则**：
   - 统一分配/释放位置：在probe中分配的资源，必须在remove中释放；在open中分配的，必须在release中释放；
   - 分支释放：用goto语句统一处理错误分支的释放逻辑，避免重复代码；
   - 动态成员释放：结构体中的动态分配成员（如`dev->buf = <span class="green">kmalloc</span>(...)`），需在释放结构体前单独释放。
   ```c
   // 优化后代码：无泄漏
   static int sensor_probe(struct i2c_client *client) {
       struct sensor_dev *dev = kzalloc(sizeof(struct sensor_dev), GFP_KERNEL);
       int ret;

       if (!dev) return -ENOMEM;

       // 初始化I2C客户端数据
       i2c_set_clientdata(client, dev);

       // 分配批量缓冲区（动态成员）
       dev->batch_buf = kmalloc(sizeof(struct batch_data), GFP_KERNEL);
       if (!dev->batch_buf) {
           ret = -ENOMEM;
           goto err_batch;
       }

       // 注册字符设备
       ret = alloc_chrdev_region(&dev->dev_num, 0, 1, "sht30");
       if (ret < 0) {
           dev_err(&client->dev, "Alloc dev num failed!\n");
           goto err_chrdev;
       }

       cdev_init(&dev->cdev, &sensor_fops);
       ret = cdev_add(&dev->cdev, dev->dev_num, 1);
       if (ret < 0) {
           dev_err(&client->dev, "Cdev add failed!\n");
           goto err_cdev;
       }

       return 0;

       // 错误分支统一释放（从后往前释放）
   err_cdev:
       unregister_chrdev_region(&dev->dev_num, 1);
   err_chrdev:
       kfree(dev->batch_buf);
   err_batch:
       kfree(dev);
       return ret;
   }

   static void sensor_remove(struct i2c_client *client) {
       struct sensor_dev *dev = i2c_get_clientdata(client);
       cdev_del(&dev->cdev);
       unregister_chrdev_region(&dev->dev_num, 1);
       kfree(dev->batch_buf);  // 释放动态成员
       kfree(dev);             // 释放结构体
   }
   ```

2.  **静态检查工具：sparse**：
   内核自带的静态检查工具，可检测“未初始化变量”“内存泄漏风险”等问题：
   ```bash
   # 1. 安装sparse
   sudo apt-get install sparse

   # 2. 编译驱动时启用sparse检查
   make C=1 CHECK=sparse -f Makefile.sensor

   # 3. 典型泄漏风险提示（示例）
   sensor_sht30_drv.c:123:3: warning: potential memory leak: dev
   ```

###### 2.1.3 动态检测：kmemleak工具实战 [M]
kmemleak是内核自带的内存泄漏检测工具，通过跟踪<span class="green">kmalloc</span>/kfree的调用，识别未释放的内存块，大师级开发者需熟练使用其定位复杂泄漏问题。

1.  **启用kmemleak（内核配置与启动）**：
   ```bash
   # 1. 配置内核（make menuconfig）
   Kernel hacking → Memory Debugging → Configure kmemleak (EXPERIMENTAL)
   # 选择"Enable kmemleak"和"kmemleak default mode"（默认开启）

   # 2. 重新编译内核并烧录
   make zImage -j8
   make dtbs

   # 3. 启动开发板，查看kmemleak状态
   cat /sys/kernel/debug/kmemleak
   # 输出：kmemleak is enabled
   ```

2.  **定位泄漏点**：
   ```bash
   # 1. 加载存在泄漏的驱动
   insmod sensor_sht30_drv.ko

   # 2. 触发泄漏场景（如probe后remove，模拟设备插拔）
   rmmod sensor_sht30_drv.ko
   insmod sensor_sht30_drv.ko

   # 3. 强制kmemleak扫描并报告泄漏
   echo scan > /sys/kernel/debug/kmemleak
   cat /sys/kernel/debug/kmemleak

   # 4. 典型泄漏报告（示例）
   unreferenced object 0xc0a00000 (size 64):
     comm "insmod", pid 1234, jiffies 1234567 (age 30 sec)
     hex dump (first 32 bytes):
       00 00 00 00 00 00 00 00 00 00 00 00 00 00 00 00  ................
       00 00 00 00 00 00 00 00 00 00 00 00 00 00 00 00  ................
     backtrace:
       [<c010a234>] kmalloc+0x44/0xc0
   ```

### <strong>句柄管理：设备文件描述符的复用与释放逻辑 [M]</strong>

文件描述符（fd）是应用层访问设备的句柄，Linux系统默认单个进程最大fd数为1024，若应用频繁打开设备未关闭，会导致“Too many open files”错误。大师级管理需实现“fd复用池”和“自动回收机制”，适配多设备<span class="red">并发</span>访问场景。

###### 2.2.1 句柄泄漏场景与危害
```c
// 泄漏代码：循环中打开设备未关闭
void read_sensor_loop(void) {
    int ret;
    struct sensor_data data;

    while (1) {
        // 泄漏点：每次循环打开fd，未关闭
        int fd = open("/dev/sht30_sensor", O_RDONLY);
        if (fd < 0) { perror("open"); continue; }

        ret = read(fd, &data, sizeof(data));
        if (ret < 0) { perror("read"); }

        // 忘记关闭fd
        printf("Temp: %.1f℃\n", data.temp);
        sleep(1);
    }
}
```
- 危害：运行1024秒后，fd达到上限，无法打开新设备：
  ```bash
  open /dev/sht30_sensor failed! errno=24: Too many open files
  ```

###### 2.2.2 管理方案：fd复用与自动回收
1.  **基础方案：单例fd（单个设备复用）**：
   应用启动时打开fd，全程复用，退出时关闭，适用于“单设备单线程”场景：
   ```c
   int g_sensor_fd = -1;  // 全局fd，单例复用

   // 初始化fd（启动时调用一次）
   int sensor_init(void) {
       if (g_sensor_fd >= 0) return 0;  // 已打开，直接返回
       g_sensor_fd = open("/dev/sht30_sensor", O_RDONLY);
       return g_sensor_fd < 0 ? -1 : 0;
   }

   // 读取数据（复用fd）
   int sensor_read_data(struct sensor_data *data) {
       if (g_sensor_fd < 0) return -1;
       return read(g_sensor_fd, data, sizeof(*data));
   }

   // 释放fd（退出时调用）
   void sensor_deinit(void) {
       if (g_sensor_fd >= 0) {
           close(g_sensor_fd);
           g_sensor_fd = -1;
       }
   }

   int main(void) {
       struct sensor_data data;
       atexit(sensor_deinit);  // 注册退出时自动关闭fd

       if (sensor_init() < 0) { perror("init"); return -1; }

       while (1) {
           if (sensor_read_data(&data) >= 0) {
               printf("Temp: %.1f℃\n", data.temp);
           }
           sleep(1);
       }
       return 0;
   }
   ```

2.  **大师级方案：fd复用池（多设备<span class="red">并发</span>）**：
   适用于“多传感器、多线程<span class="red">并发</span>访问”场景，用哈希表管理“设备路径→fd”映射，实现fd复用，避免重复打开；同时设置超时机制，自动回收长期未使用的fd。
   ```c
   #include <stdio.h>
   #include <fcntl.h>
   #include <unistd.h>
   #include <pthread.h>
   #include <time.h>
   #include <stdlib.h>
   #include <string.h>

   #define MAX_FD_COUNT 32  // 最大复用fd数
   #define TIMEOUT 30       // 超时时间（秒，未使用则回收）

   // fd复用池节点
   typedef struct fd_node {
       char dev_path[64];    // 设备路径（键）
       int fd;               // 复用的fd
       time_t last_used;     // 最后使用时间
       struct fd_node *next; // 链表节点
   } fd_node_t;

   static fd_node_t *fd_pool = NULL;
   static pthread_mutex_t pool_mutex = PTHREAD_MUTEX_INITIALIZER;

   // 从复用池获取fd（不存在则打开，存在则复用）
   int fd_pool_get(const char *dev_path, int flags) {
       fd_node_t *node = NULL;
       time_t now = time(NULL);

       pthread_mutex_lock(&pool_mutex);

       // 1. 查找设备是否已在池中有fd
       for (node = fd_pool; node; node = node->next) {
           if (strcmp(node->dev_path, dev_path) == 0) {
               // 更新最后使用时间
               node->last_used = now;
               pthread_mutex_unlock(&pool_mutex);
               return node->fd;
           }
       }

       // 2. 池满时回收超时fd
       if (fd_pool_count() >= MAX_FD_COUNT) {
           fd_node_t *prev = NULL, *timeout_node = NULL;
           for (node = fd_pool; node; prev = node, node = node->next) {
               if (now - node->last_used > TIMEOUT) {
                   timeout_node = node;
                   break;
               }
           }
           if (timeout_node) {
               close(timeout_node->fd);
               if (prev) prev->next = timeout_node->next;
               else fd_pool = timeout_node->next;
               free(timeout_node);
           } else {
               pthread_mutex_unlock(&pool_mutex);
               return -EMFILE;  // 池满且无超时fd
           }
       }

       // 3. 打开新fd并加入池
       int fd = open(dev_path, flags);
       if (fd < 0) {
           pthread_mutex_unlock(&pool_mutex);
           return -1;
       }
       node = malloc(sizeof(fd_node_t));
       strncpy(node->dev_path, dev_path, sizeof(node->dev_path)-1);
       node->fd = fd;
       node->last_used = now;
       node->next = fd_pool;
       fd_pool = node;

       pthread_mutex_unlock(&pool_mutex);
       return fd;
   }

   // 释放fd（放回复用池，不实际关闭）
   void fd_pool_put(const char *dev_path) {
       fd_node_t *node = NULL;
       time_t now = time(NULL);

       pthread_mutex_lock(&pool_mutex);
       for (node = fd_pool; node; node = node->next) {
           if (strcmp(node->dev_path, dev_path) == 0) {
               node->last_used = now;  // 更新使用时间，标记为可用
               break;
           }
       }
       pthread_mutex_unlock(&pool_mutex);
   }

   // 清理复用池（程序退出时调用）
   void fd_pool_cleanup(void) {
       fd_node_t *node = fd_pool;
       while (node) {
           fd_node_t *tmp = node;
           close(tmp->fd);
           node = node->next;
           free(tmp);
       }
       fd_pool = NULL;
   }

   // 测试：多线程并发访问多个传感器
   #define THREAD_COUNT 4
   #define DEV_COUNT 3
   const char *dev_paths[DEV_COUNT] = {
       "/dev/sht30_sensor", "/dev/ov5640_frame", "/dev/gpio_led"
   };

   void *thread_func(void *arg) {
       int dev_idx = *(int*)arg;
       const char *dev_path = dev_paths[dev_idx];
       int fd, ret;
       char buf[32];

       while (1) {
           // 从复用池获取fd（复用或打开）
           fd = fd_pool_get(dev_path, O_RDWR);
           if (fd < 0) {
               perror("fd_pool_get");
               sleep(1);
               continue;
           }

           // 模拟读写操作
           ret = read(fd, buf, sizeof(buf));
           printf("Thread %d: read %s, fd=%d, ret=%d\n",
                  pthread_self(), dev_path, fd, ret);

           // 放回复用池（不关闭）
           fd_pool_put(dev_path);
           sleep(2);
       }
       return NULL;
   }

   int main(void) {
       pthread_t threads[THREAD_COUNT];
       int dev_idx[THREAD_COUNT] = {0,1,2,0};  // 4个线程访问3个设备

       atexit(fd_pool_cleanup);  // 退出时清理池
       for (int i=0; i<THREAD_COUNT; i++) {
           pthread_create(&threads[i], NULL, thread_func, &dev_idx[i]);
       }
       for (int i=0; i<THREAD_COUNT; i++) {
           pthread_join(threads[i], NULL);
       }
       return 0;
   }
   ```

3.  **fd泄漏检测与排查**：

   ```bash
   # 2. 查看fd对应的设备/文件
   ```
